03) Дан массив из n+1 натурального числа в диапазоне [1..n]. Найдите любой
повторяющийся элемент за O(n) времени и O(1) дополнительной памяти.
    var mark = 0;
    for (var i = 1; i <= n + 1; ++i) {
        if (a[i] > 0) {
            var x = a[i];
            while (a[x] > 0) {
                var y = x;
                x = a[x];
                a[y] = mark;
            }
            if (a[x] != mark)
                return x;
            mark = mark - 1;
        }
    }

10) Как вычислить n-ое число Фибоначчи за сублинейное количество
арифметических операций?
    F1 = 1
    F0 = 0
    Fn+1 = Fn   + Fn-1;
    Fn   = Fn-1 + Fn-2;

    { Fn+1 }` = { Fn   }` { 1 1 }
    { Fn   }  = { Fn-1 }  { 1 0 }

    int fib(int n) {
        if (n > 0) {
            { Fn+1, Fn } = { 1, 0 } *
                pow({ 1, 1 }
                    { 1, 0 }, n);
            return Fn;
        }
        return 0;
    }

    int pow(int[,] a, int n) {
        if (n > 0) {
            if (n mod 2 = 0)
                return pow(a * a, n / 2);
            return a * pow(a, n - 1);
        }
        return new {
            { 1, 0 },
            { 0, 1 }
        };
    }
